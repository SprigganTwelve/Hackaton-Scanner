AUTHENTIFICATION:
controller: AuthController
route: /api/auth:
        [POST]  /login: Use for logging an user
                    input: { email: string, password: string } 
                    output: {success: bool, message: string, token?: string }  #token is not null when operation succeed

        [POST]      /logout: Use for logout a nuser
                    input:
                        header: {Authorization : 'Bearer ${token}'}
                    output:
                        { 
                            sucess: false,
                            message: "Token invalide"
                        }

        [POST] /register: 
                    input:
                        {            
                            name: string,
                            email: string,
                            password: string,
                            git_url: string,
                            git_access_token: string
                        }
                    output:
                        {
                            success: bool,
                            message: string,
                            user?: {
                                id: string,
                                email: string,
                                name:string,
                            }
                        }
